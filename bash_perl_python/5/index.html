                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create a map

  2. declare -A ages=([dad]=42 [mom]=40 [me]=7)

  3. 

  4. #2.Print all the keys and values in the map

  5. echo ${!ages[@]} 

  6. echo ${ages[@]}

  7. 

  8. -------output----------



  1. #1.Create a map

  2. 

  3. #2.Print all the keys and values in the map

  6. print join(" ", values %ages)."\n";

  7.

  6. print ages.values()

  7. 

  8.

  2. 

  3. #2.Print all the keys and values in the map

Map: Iterate over a map 

		  

		  
		  

		  
Tasks:
1.Create a map
2.Iterate over the map
		  

		       
          
          

             
             

		     

                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create a map

  2. declare -A ages=([dad]=42 [mom]=40 [me]=7)

  3. 

  4. #2.Iterate over the map

  5. for key in ${!ages[@]}; do 

  6.   echo "$key ${ages[$key]}"

  7. done

  8. 

  9. -------output----------



  1. #1.Create a map

  2. 

  3. #2.Iterate over the map

  4. mom 40

  5. dad 42

  6. me 7



                
  1. #1.Create a map

  2. %ages=(dad=>42,mom=>40,me=>7);

  3. 

  4. #2.Iterate over the map

  5. #  use foreach

  6. foreach $key (keys %ages){print "$key $ages{$key}\n"}

  7. #  or use while

  8. while (($key,$value)=each(%ages)){print "$key $value\n"}

  9. 

  10. -------output----------



  1. #1.Create a map

  2. 

  3. #2.Iterate over the map

  4. #  use foreach

  5. me 7

  6. dad 42

  7. mom 40

  8. #  or use while

  9. me 7

  10. dad 42

  11. mom 40



                
  1. #1.Create a map

  2. ages={'dad':42,'mom':48,'me':7}

  3. 

  4. #2.Iterate over the map

  5. for key in ages:

  6.     print key, ages[key]

  7. #  or

  8. for key,value in ages.items():

  9.     print key, value

  10. 

  11. 

  12. -------output----------



  1. #1.Create a map 

  2. 

  3. #2.Iterate over the map 

  4. dad 42

  5. me 7

  6. mom 48

  7. #  or 

  8. dad 42

  9. me 7

  10. mom 48



              

             
            
 
          

          

		  

        
          

		  
Map: Delete an element in a map 

		  

		  
		  

		  
Tasks:
1.Create a map
2.Remove the element  with the key of 'me'
		  

		       
          
          

             
             

		     

                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create a map

  2. declare -A ages=([dad]=42 [mom]=40 [me]=7)

  3. 

  4. #2.Remove the element  with the key of 'me'

  5. unset ages[me]

  6. declare -p ages

  7. 

  8. -------output----------



  1. #1.Create a map

  2. 

  3. #2.Remove the element  with the key of 'me'

  4. declare -A ages='([mom]="40" [dad]="42" )'



                
  1. #1.Create a map

  2. %ages=(dad=>42,mom=>40,me=>7);

  3. 

  4. #2.Remove the element  with the key of 'me'

  5. delete $ages{me};

  6. print join(' ', %ages);

  7. 

  8. 

  9. -------output----------



  1. #1.Create a map

  2. 

  3. #2.Remove the element  with the key of 'me'

  4. mom 40 dad 42



                
  1. #1.Create a map

  2. ages={'dad':42,'mom':48,'me':7}

  3. 

  4. #2.Remove the element  with the key of 'me'

  5. del ages['me']

  6. print ages.items()

  7. 

  8. -------output----------



  1. #1.Create a map 

  2. 

  3. #2.Remove the element  with the key of 'me' 

  4. [('dad', 42), ('mom', 48)]



              

             
            
 
          

          

		  

        
          

		  
Map: Clear all elements in a map 

		  

		  
		  

		  
Tasks:
1.Create a map
2.Clear the map
		  

		       
          
          

             
             

		     

                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create a map

  2. declare -A ages=([dad]=42 [mom]=40 [me]=7)

  3. declare -p ages

  4. 

  5. #2.Clear the map

  6. ages=()

  7. [[ "${ages[@]}" ]] && declare -p ages || echo "ages not defined or empty"

  8. #  or use unset

  9. declare -A ages=([dad]=42 [mom]=40 [me]=7)

  10. unset ages

  11. [[ "${ages[@]}" ]] && declare -p ages || echo "ages not defined or empty"

  12. 

  13. -------output----------



  1. #1.Create a map

  2. declare -A ages='([mom]="40" [dad]="42" [me]="7" )'

  3. 

  4. #2.Clear the map

  5. ages not defined or empty

  6. #  or use unset

  7. ages not defined or empty



                
  1. #1.Create a map

  2. %ages=(dad=>42,mom=>40,me=>7);

  3. print join(' ', %ages)."\n";

  4. 

  5. #2.Clear the map

  6. %ages=();

  7. #  do not use defined(%ages1), which is only for scalar

  8. print %ages? %ages: "ages not defined or empty\n";

  9. #  or use undef

  10. undef %ages;

  11. print %ages? %ages: "ages not defined or empty\n";

  12. 

  13. 

  14. -------output----------



  1. #1.Create a map

  2. mom 40 dad 42 me 7

  3. 

  4. #2.Clear the map

  5. #  do not use defined(%ages1), which is only for scalar

  6. ages not defined or empty

  7. #  or use undef

  8. ages not defined or empty



                
  1. #1.Create a map

  2. ages={'dad':42,'mom':48,'me':7}

  3. print ages

  4. 

  5. #2.Clear the map

  6. ages.clear()

  7. print ages if 'ages' in locals() else "ages not defined"

  8. #  or use del

  9. del ages

  10. print ages if 'ages' in locals() else "ages not defined"

  11. 

  12. -------output----------



  1. #1.Create a map 

  2. {'dad': 42, 'me': 7, 'mom': 48}

  3. 

  4. #2.Clear the map 

  5. {}

  6. #  or use del 

  7. ages not defined
