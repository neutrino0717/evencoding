Set: Check membership of a value in a set 

In Bash, set data structure is not support , however in bash 4, associative array can be used to simulate set

In Perl, there is no native data type for set too, we can use hash with dummy value or some cpan modules such as Set::Scalar instead.

In Python, set is supported well. 

Tasks:
1.Create a set from elements with duplicates
2.Print the set
3.Check the membership of '0'
Bash Perl Python

  2. arr=(0 2 4 6 8 2 4 6)

  3. declare -A set1

  4. for i in "${arr[@]}"; do set1[$i]=1; done

  6. #2.Print the set

  7. #  use 'declare -p' to print

  8. declare -p set1

  9. #  or use for loop

  10. for i in "${!set1[@]}"; do echo -n "$i "; done; echo

  11. 

  12. #3.Check the membership of '0'

  13. [ -n "${set1[0]}" ] && echo "found 0 in set"

  14.

Set: Get intersection of two sets 

		  

		  
		  

		  
Tasks:
1.Create two sets
2.Get intersection of the two sets
		  

		       
          
          

             
             

		     

                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create two sets

  2. declare -A set1='([0]="1" [2]="1" [4]="1" [6]="1" [8]="1" )'

  3. declare -A set2='([0]="1" [3]="1" [6]="1" [9]="1" )'

  4. 

  5. #2.Get intersection of the two sets

  6. oldIFS=$IFS IFS=$'\n\t'

  7. arri=($(comm -12  <(echo "${!set1[*]}"|sort) <(echo "${!set2[*]}"|sort)))

  8. IFS=$oldIFS

  9. declare -A seti

  10. for i in "${arri[@]}"; do seti[$i]=1; done

  11. declare -p seti

  12. 

  13. 

  14. -------output----------



  1. #1.Create two sets

  2. 

  3. #2.Get intersection of the two sets

  4. declare -A seti='([0]="1" [6]="1" )'



                
  1. #1.Create two sets

  2. %set1 = map { $_ => 1 }(0,2,4,6,8);

  3. %set2 = map { $_ => 1 }(0,3,6,9);

  4. 

  5. #2.Get intersection of the two sets

  6. #  for each $_ in %set1, lookup it in $set2, if found,it's eligable

  7. %seti = map { $_ => 1 } grep { $set2{$_} } keys %set1;

  8. print "$_ " foreach(keys %seti);

  9. 

  10. 

  11. -------output----------

  2. 

  3. #2.Get intersection of the two sets

  1. #1.Create two sets

  2. set1 = set([0,2,4,6,8])

  3. set2 = set((0,3,6,9))

  4. 

  5. #2.Get intersection of the two sets

  6. print set1 & set2

  7. #  or use intersection() function

  8. print set1.intersection(set2)

  9. 

  10. -------output----------

Set: Get union of two sets 

		  

		  
		  

		  
Tasks:
1.Create two sets
2.Get union of the two sets
		  

		       
          
          

             
             

		     

                Bash
                Perl
                Python			   
		     

             
              

                
  1. #1.Create two sets

  2. declare -A set1='([0]="1" [2]="1" [4]="1" [6]="1" [8]="1" )'

  3. declare -A set2='([0]="1" [3]="1" [6]="1" [9]="1" )'

  4. 

  5. #2.Get union of the two sets

  6. oldIFS=$IFS IFS=$'\n\t'

  7. arru=($(comm  <(echo "${!set1[*]}"|sort) <(echo "${!set2[*]}"|sort)))

  8. IFS=$oldIFS

  9. declare -A setu

  10. for i in "${arru[@]}"; do setu[$i]=1; done

  11. declare -p setu

  12. 

  13. 

  14. -------output----------



  1. #1.Create two sets

  2. 

  3. #2.Get union of the two sets

  4. declare -A setu='([0]="1" [2]="1" [3]="1" [4]="1" [6]="1" [8]="1" [9]="1" )'



                
  1. #1.Create two sets

  2. %set1 = map { $_ => 1 }(0,2,4,6,8);

  3. %set2 = map { $_ => 1 }(0,3,6,9);

  4. 

  5. #2.Get union of the two sets

  6. #  merge the two array from keys of set1 and set2

  7. %setu = map { $_ => 1 } keys %set1, keys %set2;

  8. print "$_ " foreach(keys %setu);

  9. 

  10. -------output----------

  2.

  4. 

  5. #2.Get union of the two sets

  6. print set1 | set2

  7. #  or use union() function

  8. print set1.union(set2)

  9.
